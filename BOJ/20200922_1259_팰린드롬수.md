### 문제

[팰린드롬수  백준  1259번](https://www.acmicpc.net/problem/1259)

### 맞은 코드 

```c++
#include <iostream>
#include <string>
#include <vector>
using namespace std;
 
int n;

int main(void){
 
	while(1){
		vector<int> v;
		string st = "yes";
		int len = 0;
		
		cin >> n;
		if(n==0) break;
		
		while(n > 0){
			v.push_back(n % 10);	
			n /= 10;
			len++;
		}
		 		
		len--;
		
		for(int i = 0; i <= len/2; i++){
			if(v[i] != v[len-i]) st = "no";
		}
		cout << st << '\n';
		
	}
	
	return 0;
} 
```
